00069         raise Exception("Depth image is not in intensity format")

00074         for u in range(0,rgb.size[0],downsample):
00075             color = rgb.getpixel((u,v))

00077             if Z==0: continue

00079             Y = (v - centerY) * Z / focalLength

00084               vec_transf = numpy.dot(transform,vec_org)
00085               points.append("%f %f %f %d %d %d 0\n"%(vec_transf[0,0],vec_transf[1,0],vec_transf[2,0],color[0],color[1],color[2]))
